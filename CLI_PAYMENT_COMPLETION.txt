## STEP 6: Polar Webhook Handler ✅

**File**: `src/app/api/webhooks/polar/route.ts`

**Requirements**:
- Verify webhook signature (security critical)
- Handle events: `checkout.created`, `checkout.completed`, `subscription.created`, `subscription.updated`, `subscription.canceled`
- Update Supabase tables: `transactions`, `subscriptions`, `users`
- Send confirmation emails via Resend
- Error handling & logging

**Implementation**:
```typescript
import { NextRequest, NextResponse } from 'next/server';
import { Polar } from '@polar-sh/sdk';
import { getSupabaseClient } from '@/lib/supabase';
import { sendEmail } from '@/lib/email-service';
import { emailTemplates } from '@/lib/email-templates';
import crypto from 'crypto';

const polar = new Polar({
  accessToken: process.env.POLAR_ACCESS_TOKEN!,
});

export async function POST(req: NextRequest) {
  try {
    // 1. Verify webhook signature
    const signature = req.headers.get('polar-signature');
    const body = await req.text();
    
    if (!signature || !verifySignature(body, signature, process.env.POLAR_WEBHOOK_SECRET!)) {
      return NextResponse.json({ error: 'Invalid signature' }, { status: 401 });
    }

    const event = JSON.parse(body);
    const supabase = getSupabaseClient();

    // 2. Handle different event types
    switch (event.type) {
      case 'checkout.completed':
        const checkout = event.data;
        
        // Create transaction record
        const { error: txError } = await supabase.from('transactions').insert({
          user_id: checkout.customer_id || checkout.metadata?.userId,
          amount: checkout.amount / 100, // Convert from cents
          currency: checkout.currency || 'USD',
          status: 'completed',
          provider: 'polar',
          provider_transaction_id: checkout.id,
          metadata: checkout,
          created_at: new Date().toISOString(),
        });

        if (txError) {
          console.error('Transaction insert error:', txError);
        }

        // Get tier from product metadata
        const tier = checkout.product?.metadata?.tier || checkout.metadata?.tier || 'pro';

        // Update user subscription
        const { error: userError } = await supabase.from('users').update({
          subscription_tier: tier,
          subscription_status: 'active',
          subscription_price: checkout.amount / 100,
          subscription_expires_at: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000).toISOString(),
        }).eq('id', checkout.customer_id || checkout.metadata?.userId);

        if (userError) {
          console.error('User update error:', userError);
        }

        // Send confirmation email
        if (checkout.customer_email) {
          await sendEmail({
            to: checkout.customer_email,
            subject: '🎉 Payment Successful - Welcome to ApexOS!',
            html: `
              <h1>Thank you for your purchase!</h1>
              <p>Your ${tier.toUpperCase()} subscription is now active.</p>
              <p>Amount: $${(checkout.amount / 100).toFixed(2)}</p>
              <p>Start using ApexOS: <a href="https://apexrebate.com/dashboard">Dashboard</a></p>
            `,
          });
        }
        break;

      case 'subscription.created':
      case 'subscription.updated':
        const sub = event.data;
        
        await supabase.from('subscriptions').upsert({
          id: sub.id,
          user_id: sub.customer_id || sub.metadata?.userId,
          tier: sub.product?.metadata?.tier || 'pro',
          status: sub.status,
          current_period_start: sub.current_period_start,
          current_period_end: sub.current_period_end,
          cancel_at_period_end: sub.cancel_at_period_end || false,
        });
        break;

      case 'subscription.canceled':
        await supabase.from('subscriptions').update({
          status: 'canceled',
          canceled_at: new Date().toISOString(),
        }).eq('id', event.data.id);

        // Update user status
        await supabase.from('users').update({
          subscription_status: 'canceled',
        }).eq('id', event.data.customer_id || event.data.metadata?.userId);
        break;
    }

    return NextResponse.json({ received: true });
  } catch (error) {
    console.error('Polar webhook error:', error);
    return NextResponse.json({ error: 'Webhook processing failed' }, { status: 500 });
  }
}

function verifySignature(body: string, signature: string, secret: string): boolean {
  const expectedSignature = crypto
    .createHmac('sha256', secret)
    .update(body)
    .digest('hex');
  return crypto.timingSafeEqual(
    Buffer.from(signature),
    Buffer.from(expectedSignature)
  );
}
```

**File**: `src/app/api/webhooks/nowpayments/route.ts`

**Requirements**:
- Verify IPN signature
- Handle payment status updates
- Update database on payment completion
- Send confirmation emails

**Implementation**:
```typescript
import { NextRequest, NextResponse } from 'next/server';
import { getSupabaseClient } from '@/lib/supabase';
import { sendEmail } from '@/lib/email-service';
import crypto from 'crypto';

export async function POST(req: NextRequest) {
  try {
    // 1. Verify IPN signature
    const signature = req.headers.get('x-nowpayments-sig');
    const body = await req.text();
    const event = JSON.parse(body);

    // NOWPayments HMAC verification
    const expectedSignature = crypto
      .createHmac('sha512', process.env.NOWPAYMENTS_IPN_SECRET!)
      .update(JSON.stringify(event))
      .digest('hex');

    if (signature !== expectedSignature) {
      console.error('Invalid NOWPayments signature');
      return NextResponse.json({ error: 'Invalid signature' }, { status: 401 });
    }

    const supabase = getSupabaseClient();

    // 2. Handle payment status
    if (event.payment_status === 'finished') {
      // Extract user_id from order_id (format: userId_tier_timestamp)
      const [userId, tier] = event.order_id.split('_');

      // Create transaction record
      await supabase.from('transactions').insert({
        user_id: userId,
        amount: parseFloat(event.price_amount),
        currency: event.price_currency,
        status: 'completed',
        provider: 'nowpayments',
        provider_transaction_id: event.payment_id,
        metadata: {
          crypto_amount: event.pay_amount,
          crypto_currency: event.pay_currency,
          actually_paid: event.actually_paid,
        },
        created_at: new Date().toISOString(),
      });

      // Update user subscription
      await supabase.from('users').update({
        subscription_tier: tier,
        subscription_status: 'active',
        subscription_price: parseFloat(event.price_amount),
        subscription_expires_at: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000).toISOString(),
      }).eq('id', userId);

      // Get user email
      const { data: user } = await supabase
        .from('users')
        .select('email')
        .eq('id', userId)
        .single();

      // Send confirmation email
      if (user?.email) {
        await sendEmail({
          to: user.email,
          subject: '🎉 Crypto Payment Received - Welcome to ApexOS!',
          html: `
            <h1>Payment Confirmed!</h1>
            <p>Your ${tier.toUpperCase()} subscription is now active.</p>
            <p>Amount paid: ${event.pay_amount} ${event.pay_currency}</p>
            <p>Transaction ID: ${event.payment_id}</p>
            <p>Start using ApexOS: <a href="https://apexrebate.com/dashboard">Dashboard</a></p>
          `,
        });
      }
    } else if (event.payment_status === 'failed' || event.payment_status === 'expired') {
      // Log failed payment
      console.error('Payment failed:', event);
      
      // Optional: Notify user
      const [userId] = event.order_id.split('_');
      const { data: user } = await supabase
        .from('users')
        .select('email')
        .eq('id', userId)
        .single();

      if (user?.email) {
        await sendEmail({
          to: user.email,
          subject: '❌ Payment Failed - ApexOS',
          html: `
            <h1>Payment Failed</h1>
            <p>Your payment could not be processed.</p>
            <p>Status: ${event.payment_status}</p>
            <p>Please try again: <a href="https://apexrebate.com/pricing">Pricing Page</a></p>
          `,
        });
      }
    }

    return NextResponse.json({ success: true });
  } catch (error) {
    console.error('NOWPayments webhook error:', error);
    return NextResponse.json({ error: 'Webhook processing failed' }, { status: 500 });
  }
}
```

**Implementation**:
```typescript
import { NextRequest, NextResponse } from 'next/server';
import { Polar } from '@polar-sh/sdk';
import { getSupabaseClient } from '@/lib/supabase';
import { trackEvent } from '@/lib/analytics';

const polar = new Polar({ accessToken: process.env.POLAR_ACCESS_TOKEN! });

// Tier pricing configuration
const TIER_PRICES = {
  pro: { usd: 29, name: 'Pro' },
  trader: { usd: 97, name: 'Trader' },
  elite: { usd: 297, name: 'Elite' },
};

export async function POST(req: NextRequest) {
  try {
    const { tier, paymentMethod, userId } = await req.json();
    const supabase = getSupabaseClient();

    // Validate inputs
    if (!tier || !paymentMethod || !userId) {
      return NextResponse.json({ error: 'Missing required fields' }, { status: 400 });
    }

    // Validate user
    const { data: user, error: userError } = await supabase
      .from('users')
      .select('email, id')
      .eq('id', userId)
      .single();

    if (userError || !user) {
      return NextResponse.json({ error: 'User not found' }, { status: 404 });
    }

    // Get tier pricing
    const selectedTier = TIER_PRICES[tier as keyof typeof TIER_PRICES];
    if (!selectedTier) {
      return NextResponse.json({ error: 'Invalid tier' }, { status: 400 });
    }

    let checkoutUrl = '';

    if (paymentMethod === 'card') {
      // ✅ Use Polar for card payments
      const productPriceId = process.env[`POLAR_${tier.toUpperCase()}_PRICE_ID`];
      if (!productPriceId) {
        return NextResponse.json({ error: 'Product price not configured' }, { status: 500 });
      }

      const checkout = await polar.checkouts.create({
        productPriceId,
        customerEmail: user.email,
        successUrl: `${process.env.NEXT_PUBLIC_BASE_URL}/checkout/success`,
        metadata: { userId: user.id, tier },
      });
      
      checkoutUrl = checkout.url;

    } else if (paymentMethod === 'crypto') {
      // ✅ Use NOWPayments for crypto
      const orderId = `${user.id}_${tier}_${Date.now()}`;
      
      const nowPaymentsResponse = await fetch('https://api.nowpayments.io/v1/payment', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'x-api-key': process.env.NOWPAYMENTS_API_KEY!,
        },
        body: JSON.stringify({
          price_amount: selectedTier.usd,
          price_currency: 'usd',
          pay_currency: 'btc', // User can choose on their end
          order_id: orderId,
          order_description: `ApexOS ${selectedTier.name} Subscription`,
          ipn_callback_url: `${process.env.NEXT_PUBLIC_BASE_URL}/api/webhooks/nowpayments`,
          success_url: `${process.env.NEXT_PUBLIC_BASE_URL}/checkout/success`,
          cancel_url: `${process.env.NEXT_PUBLIC_BASE_URL}/pricing`,
        }),
      });

      const nowPaymentsData = await nowPaymentsResponse.json();
      checkoutUrl = nowPaymentsData.invoice_url;

    } else {
      return NextResponse.json({ error: 'Invalid payment method' }, { status: 400 });
    }

    // Track checkout event
    await trackEvent({
      event_name: 'checkout_started',
      user_id: userId,
      metadata: { tier, paymentMethod, amount: selectedTier.usd },
    });

    return NextResponse.json({ checkoutUrl });
  } catch (error) {
    console.error('Checkout API error:', error);
    return NextResponse.json({ 
      error: 'Checkout failed', 
      details: error instanceof Error ? error.message : 'Unknown error' 
    }, { status: 500 });
  }
}
```

```typescript
'use client';

import { useState } from 'react';
import { PaymentMethodSelector } from './PaymentMethodSelector';
import { Button } from '@/components/ui/button';
import { X, Loader2 } from 'lucide-react';

interface CheckoutModalProps {
  tier: 'pro' | 'trader' | 'elite';
  userId: string;
  onClose: () => void;
  isOpen: boolean;
}

const TIER_DETAILS = {
  pro: { price: 29, name: 'Pro' },
  trader: { price: 97, name: 'Trader' },
  elite: { price: 297, name: 'Elite' },
};

export function CheckoutModal({ tier, userId, onClose, isOpen }: CheckoutModalProps) {
  const [paymentMethod, setPaymentMethod] = useState<'card' | 'crypto' | null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  if (!isOpen) return null;

  const tierDetails = TIER_DETAILS[tier];

  const handleCheckout = async () => {
    if (!paymentMethod) return;

    setLoading(true);
    setError(null);

    try {
      const response = await fetch('/api/checkout', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ tier, paymentMethod, userId }),
      });

      if (!response.ok) {
        const data = await response.json();
        throw new Error(data.error || 'Checkout failed');
      }

      const { checkoutUrl } = await response.json();
      
      if (!checkoutUrl) {
        throw new Error('No checkout URL received');
      }

      // Redirect to payment page
      window.location.href = checkoutUrl;
    } catch (err) {
      console.error('Checkout error:', err);
      setError(err instanceof Error ? err.message : 'Checkout failed. Please try again.');
      setLoading(false);
    }
  };

  return (
    <div className="fixed inset-0 bg-black/70 backdrop-blur-sm flex items-center justify-center z-50 p-4">
      <div className="bg-[#0a0a0a] border border-white/10 rounded-lg max-w-md w-full p-8 relative">
        {/* Close button */}
        <button
          onClick={onClose}
          className="absolute top-4 right-4 text-zinc-400 hover:text-white transition"
          disabled={loading}
        >
          <X className="w-5 h-5" />
        </button>

        {/* Header */}
        <div className="mb-6">
          <h2 className="text-2xl font-bold mb-2">Upgrade to {tierDetails.name}</h2>
          <p className="text-3xl font-bold text-emerald-400">${tierDetails.price}<span className="text-lg text-zinc-400">/month</span></p>
        </div>

        {/* Payment method selector */}
        <div className="mb-6">
          <h3 className="text-sm font-medium text-zinc-400 mb-3">Choose Payment Method</h3>
          <PaymentMethodSelector onSelect={setPaymentMethod} />
        </div>

        {/* Error message */}
        {error && (
          <div className="mb-4 p-3 bg-red-500/10 border border-red-500/20 rounded text-red-400 text-sm">
            {error}
          </div>
        )}

        {/* Action buttons */}
        <div className="flex gap-3">
          <Button
            onClick={onClose}
            variant="outline"
            disabled={loading}
            className="flex-1"
          >
            Cancel
          </Button>
          <Button
            onClick={handleCheckout}
            disabled={!paymentMethod || loading}
            className="flex-1 bg-emerald-500 hover:bg-emerald-600"
          >
            {loading ? (
              <>
                <Loader2 className="w-4 h-4 mr-2 animate-spin" />
                Processing...
              </>
            ) : (
              'Continue to Payment'
            )}
          </Button>
        </div>
      </div>
    </div>
  );
}
```
