EXAMPLE: Wallet Page Integration
═══════════════════════════════════════════════════════════════════

```typescript
'use client';

import { Suspense } from 'react';
import { RealtimeWallet } from '@/components/dashboard/RealtimeWallet';
import { LockedFeature } from '@/components/dashboard/LockedFeature';
import { useUser } from '@/hooks/useUser'; // or appropriate auth hook

export default function WalletPage() {
  const { user, tier } = useUser();
  
  if (tier === 'FREE') {
    return (
      <div className="max-w-7xl mx-auto">
        <h1 className="text-3xl font-bold mb-6">Wallet</h1>
        <LockedFeature 
          feature="wallet"
          requiredTier="BASIC"
          upgradeUrl="/en/pricing"
        />
      </div>
    );
  }

  return (
    <div className="max-w-7xl mx-auto">
      <h1 className="text-3xl font-bold mb-6">Wallet</h1>
      <Suspense fallback={<WalletSkeleton />}>
        <RealtimeWallet userId={user.id} />
      </Suspense>
    </div>
  );
}

function WalletSkeleton() {
  return <div className="animate-pulse">Loading...</div>;
}
```
