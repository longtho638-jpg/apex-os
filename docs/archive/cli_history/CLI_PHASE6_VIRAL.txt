## TASK 2: Commission Tracking & Payout

### 2.1 Commission Calculator (Dynamic L1-L4)
**File**: `src/lib/referral-commission.ts`

```typescript
import { UNIFIED_TIERS, getCommissionRate, TierId } from '@/config/unified-tiers';

export async function calculateCommission(
  referredUserId: string,
  subscriptionAmount: number
) {
  // 1. Find direct referrer
  const { data: conversion } = await supabase
    .from('referral_conversions')
    .select('*, referrer:users(subscription_tier)') // Need referrer's tier
    .eq('referred_user_id', referredUserId)
    .single();
  
  if (!conversion) return null;
  
  const referrerTier = conversion.referrer.subscription_tier as TierId;
  
  // 2. Calculate L1 Commission (Direct)
  // Use getCommissionRate(tier, level) from unified-tiers.ts
  const rateL1 = getCommissionRate(referrerTier, 1); 
  const commissionAmount = subscriptionAmount * rateL1;
  
  // 3. Update conversion record
  if (commissionAmount > 0) {
      await supabase
        .from('referral_conversions')
        .update({
          subscription_revenue: subscriptionAmount,
          commission_amount: commissionAmount,
        })
        .eq('id', conversion.id);
        
      // Credit Wallet (Realtime)
      await supabase.rpc('credit_user_balance_realtime', {
          p_user_id: conversion.referrer_id,
          p_amount: commissionAmount,
          p_source: 'referral_l1',
          p_metadata: { referred_user: referredUserId }
      });
  }
  
  // TODO: Implement L2-L4 recursion here if needed (Viral Loop)
  
  return { referrerId: conversion.referrer_id, commissionAmount };
}
```

**Trigger**: When subscription payment succeeds (NOWPayments/Polar webhook)
